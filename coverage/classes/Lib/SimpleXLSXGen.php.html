399    protected function _writeEntry($fh, &$cdrec, $cfilename, $data) {
400        $zipSignature = "\x50\x4b\x03\x04"; // local file header signature
401        $dirSignature = "\x50\x4b\x01\x02"; // central dir header signature

404        $e['uncsize'] = mb_strlen($data, '8bit');

406        // if data to compress is too small, just store it

411            $zdata = $data;

448        fwrite($fh, pack('s', 0));  // extra_field_length

465        $cdrec .= pack('V', $e['comsize']); // compressed filesize
466        $cdrec .= pack('V', $e['uncsize']); // uncompressed filesize
